1、maven配置文件中的配置
<!-- 用JDK1.8编译、运行项目 -->
<profiles>
	<profile>
		<id>jdk\1.8</id>
		<activation>
			<activeByDefault>true</activeByDefault>
			<jdk>1.8</jdk>
		</activation>
		<properties>
			<maven.compiler.source>1.8</maven.compiler.source>
			<maven.compiler.target>1.8</maven.compiler.target>
			<maven.compiler.compilerVersion>1.8</maven.compiler.compilerVersion>
		</properties>
	</profile>
</profiles>

2、读取配置文件的配置：
	@Component
	@ConfigurationProperties(prefix = "person")  ---》属性在主配置文件application.yml内部
	public class Person{}
	
	如果读取的属性是单独 的properties文件，使用如下注解：
	@PropertySource(value = {"classpath:person.properties"})
	public class Person{}
